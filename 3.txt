FROM Products

WHERE ProductName LIKE '%u'

SET 
	Price = CASE
	WHEN CategoryID = 1 THEN Price * .71
	WHEN CategoryID = 3 THEN Price * .005
	END
WHERE CategoryID IN (1, 3)
